CREATE TABLE BOOKING (
    BOOKING_NO NUMBER PRIMARY KEY,
    USER_ID VARCHAR2(20) NOT NULL,
    BOOKING_DATE DATE DEFAULT SYSDATE,
    BOOKING_PEO VARCHAR2(10),
    CATEGORY_NAME VARCHAR2(100) NOT NULL,
    CONTENT_NAME VARCHAR2(100) NOT NULL,
    OPTION_NAME VARCHAR2(20),
    BOOKING_PRICE VARCHAR2(20)
);

COMMENT ON COLUMN BOOKING.BOOKING_NO IS '예약번호';
COMMENT ON COLUMN BOOKING.USER_ID IS '회원아이디';
COMMENT ON COLUMN BOOKING.BOOKING_DATE IS '예약날짜';
COMMENT ON COLUMN BOOKING.BOOKING_PEO IS '예약인원';
COMMENT ON COLUMN BOOKING.CATEGORY_NAME IS '카테고리명';
COMMENT ON COLUMN BOOKING.CONTENT_NAME IS '컨텐츠명';
COMMENT ON COLUMN BOOKING.OPTION_NAME IS '옵션명';
COMMENT ON COLUMN BOOKING.BOOKING_PRICE IS '예약금액';

CREATE TABLE REVIEW (
    BOOKING_NO NUMBER PRIMARY KEY,
    CONTENT_NAME VARCHAR2(20) NOT NULL,
    REVIEW_CONTENT VARCHAR2(3000) NOT NULL,
    REVIEW_DATE DATE DEFAULT SYSDATE,
    USER_ID VARCHAR2(20)
);

COMMENT ON COLUMN REVIEW.BOOKING_NO IS '예약번호';
COMMENT ON COLUMN REVIEW.CONTENT_NAME IS '컨텐츠명';
COMMENT ON COLUMN REVIEW.REVIEW_CONTENT IS '후기내용';
COMMENT ON COLUMN REVIEW.REVIEW_DATE IS '작성일';
COMMENT ON COLUMN REVIEW.USER_ID IS '회원아이디';

CREATE TABLE INTEREST_HG (
    BOARD_NO NUMBER PRIMARY KEY,
    CONTENT_NO NUMBER,
    CONTENT_NAME VARCHAR2(70) NOT NULL,
    USER_ID VARCHAR2(20),
    I_CREATE_DATE DATE DEFAULT SYSDATE
);

COMMENT ON COLUMN INTEREST_HG.CONTENT_NO IS '컨텐츠 번호';
COMMENT ON COLUMN INTEREST_HG.USER_ID IS '회원 아이디';
COMMENT ON COLUMN INTEREST_HG.BOARD_NO IS '게시글 번호';
COMMENT ON COLUMN INTEREST_HG.I_CREATE_DATE IS '관심 목록 등록일';
COMMENT ON COLUMN INTEREST_HG.CONTENT_NAME IS '관심 컨텐츠명';

CREATE TABLE CONTENTS (
    CONTENTS_NO NUMBER PRIMARY KEY,
    CONTENTS_TYPE VARCHAR2(30) NOT NULL,
    CONTENTS_IMG VARCHAR2(300) NOT NULL,
    CONTENTS_TITLE VARCHAR2(100) NOT NULL,
    CONTENTS_DETAIL VARCHAR2(3000) NOT NULL,
    CONTENTS_DAY DATE DEFAULT SYSDATE,
    CONTENTS_VIEWS NUMBER,
    CONTENTS_WRITER VARCHAR2(300),
    CONTENTS_END DATE DEFAULT SYSDATE+30
);

COMMENT ON COLUMN CONTENTS.CONTENTS_NO IS '컨텐츠번호';
COMMENT ON COLUMN CONTENTS.CONTENTS_TYPE IS '컨텐츠타입';
COMMENT ON COLUMN CONTENTS.CONTENTS_IMG IS '컨텐츠이미지';
COMMENT ON COLUMN CONTENTS.CONTENTS_DETAIL IS '컨텐츠디테일';
COMMENT ON COLUMN CONTENTS.CONTENTS_TITLE IS '컨텐츠제목';
COMMENT ON COLUMN CONTENTS.CONTENTS_DAY IS  '컨텐츠날짜';
COMMENT ON COLUMN CONTENTS.CONTENTS_VIEWS IS '컨텐츠조회수';
COMMENT ON COLUMN CONTENTS.CONTENTS_WRITER IS '컨텐츠작성자';
COMMENT ON COLUMN CONTENTS.CONTENTS_END IS '컨텐츠종료일';

CREATE TABLE QNA (
    QNA_NO NUMBER PRIMARY KEY,
    QNA_TITLE VARCHAR(300) NOT NULL,
    QNA_CONTENT VARCHAR(3000) NOT NULL,
    QNA_WRITER VARCHAR2(30) NOT NULL,
    QNA_DATE DATE DEFAULT SYSDATE,
    ANSWER_CONTENT VARCHAR2(3000),
    ANSWER_WRITER VARCHAR2(30),
    ANSWER_DATE DATE DEFAULT SYSDATE ,

COMMENT ON COLUMN QNA.QNA_TITLE IS 'qna제목';
COMMENT ON COLUMN QNA.QNA_CONTENT IS 'qna내용';
COMMENT ON COLUMN QNA.QNA_WRITER IS 'qna작성자';

CREATE TABLE BOARD(
    BOARD_NO NUMBER PRIMARY KEY,
    BOARD_TITLE VARCHAR2(300) NOT NULL,
    BOARD_CONTENTS VARCHAR2(3000) NOT NULL,
    BOARD_WRITER  VARCHAR2(30)  NOT NULL,
    BOARD_FILENAME VARCHAR2(1000),
    BOARD_FILERENAME VARCHAR2(1000),
    BOARD_COUNT NUMBER DEFAULT 0,
    BOARD_CREATE_DATE DATE DEFAULT SYSDATE,
    BOARD_UPDATE_DATE DATE DEFAULT SYSDATE,
    B_STATUS VARCHAR2(2) DEFAULT 'Y'
);

COMMENT ON COLUMN BOARD.BOARD_NO IS '게시판번호';
COMMENT ON COLUMN BOARD.BOARD_TITLE  IS '게시판제목';
COMMENT ON COLUMN BOARD.BOARD_CONTENTS  IS '게시판내용';
COMMENT ON COLUMN BOARD.BOARD_WRITER  IS '게시판작성자';
COMMENT ON COLUMN BOARD.BOARD_FILENAME  IS '게시판파일이름';
COMMENT ON COLUMN BOARD.BOARD_FILERENAME  IS '게시판파일번호';
COMMENT ON COLUMN BOARD.BOARD_COUNT  IS '게시판조회수';
COMMENT ON COLUMN BOARD.BOARD_CREATE_DATE  IS '게시판등록일';
COMMENT ON COLUMN BOARD.BOARD_UPDATE_DATE  IS '게시판수정일';

CREATE TABLE COMMENTS(
    COMMENTS_NO NUMBER NOT NULL,
    BOARD_NO NUMBER,

    COMMENTS_DATE DATE DEFAULT SYSDATE

COMMENT ON COLUMN COMMENTS.COMMENTS_NO IS '댓글번호';
